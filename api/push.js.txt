import admin from "firebase-admin";

if (!admin.apps.length) {
  admin.initializeApp({
    credential: admin.credential.cert({
      projectId: "bustrackingsystem-62894",
      clientEmail: "firebase-adminsdk-xxxxx@bustrackingsystem-62894.iam.gserviceaccount.com",
      privateKey: process.env.FIREBASE_PRIVATE_KEY.replace(/\\n/g, "\n")
    }),
    databaseURL: "https://bustrackingsystem-62894-default-rtdb.asia-southeast1.firebasedatabase.app"
  });
}

export default async function handler(req, res) {
  if (req.method !== "POST") {
    return res.status(405).json({ error: "Only POST allowed" });
  }

  try {
    const { lat, lng, speedKmh, timestampMs } = req.body;

    await admin.database().ref("bus/location").set({
      lat,
      lng,
      speedKmh,
      timestampMs
    });

    res.status(200).json({ status: "ok" });
  } catch (err) {
    res.status(500).json({ error: err.message });
  }
}
